use peroxide::fuga::*;

fn main() -> Result<(), Box<dyn Error>> {
    let x = seq(0, 10, 1);
    let y = x.fmap(|t| t.powi(2));
    let z1 = vec![true, false];
    let z2 = vec!["a", "b"].into_iter().map(|t| t.to_string()).collect::<Vec<_>>();
    let z3 = vec!['a', 'b'];

    let mut df = DataFrame::new(vec![]);
    df.push("x", Series::new(x));
    df.push("y", Series::new(y));
    df.push("z1", Series::new(z1));
    df.push("z2", Series::new(z2));
    df.push("z3", Series::new(z3));

    df.print();

    df.write_parquet("test.parquet")?;

    let dg = DataFrame::read_parquet("test.parquet")?;
    dg.print();

    Ok(())
}
