use peroxide::fuga::*;

fn main() {
    let x = ml_matrix("1 2;3 4;5 6");
    let y = x.col_map(|c| c.fmap(|t| t - c.mean()));

    assert_eq!(y, ml_matrix("-2 -2;0 0;2 2"));
